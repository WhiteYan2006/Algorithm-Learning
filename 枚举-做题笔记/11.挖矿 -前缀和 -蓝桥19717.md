## 代码
```C++
#include <bits/stdc++.h>
using namespace std;
int main() {
  int offset = 1e6 + 1;
  int n, m;
  cin >> n >> m;
  vector<int> sum(offset * 2);
  while (n--) {  // 输入坐标
    int x;
    cin >> x;
    sum[x + offset]++;  // 用坐标为下标，要加上偏移量
  }
  partial_sum(sum.begin(), sum.end(), sum.begin());
  int ans = 0;
  for (int step = 0; step * 2 <= m; step++) {
    // 先往左走的情况
    int l = -step + offset;
    int r = m - step * 2 + offset;
    ans = max(ans, sum[r] - sum[l - 1]);
    // 先往右走的情况
    r = step + offset;
    l = -(m - step * 2) + offset;
    ans = max(ans, sum[r] - sum[l - 1]);
  }
  cout << ans << endl;
  return 0;
}
```
